#include <stdio.h>

int main() {
    char str[1000];
    int len = 0;

    // Read the string
    fgets(str, sizeof(str), stdin);

    // Find the length (excluding newline)
    while(str[len] != '\0' && str[len] != '\n') {
        len++;
    }

    // Check palindrome
    int palindrome = 1; // assume true
    for(int i = 0; i < len / 2; i++) {
        if(str[i] != str[len - 1 - i]) {
            palindrome = 0;
            break;
        }
    }

    // Print result
    if(palindrome)
        printf("Palindrome");
    else
        printf("Not palindrome");

    return 0;
}
