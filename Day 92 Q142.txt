#include <stdio.h>

// Define struct Student
struct Student {
    char name[50];
    int roll_no;
    int marks;
};

int main() {
    struct Student students[5];  // Array to store 5 students

    // Read details for 5 students
    for(int i = 0; i < 5; i++) {
        printf("Enter details for student %d:\n", i+1);

        printf("Name: ");
        scanf("%s", students[i].name);  // assumes no spaces in name

        printf("Roll: ");
        scanf("%d", &students[i].roll_no);

        printf("Marks: ");
        scanf("%d", &students[i].marks);

        printf("\n");
    }

    // Print all student details in a tabular format
    printf("Name\tRoll\tMarks\n");
    printf("---------------------------\n");
    for(int i = 0; i < 5; i++) {
        printf("%s\t%d\t%d\n", students[i].name, students[i].roll_no, students[i].marks);
    }

    return 0;
}
