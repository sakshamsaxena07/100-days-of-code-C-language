#include <stdio.h>

int main() {
    FILE *fp;
    char filename[100];
    char text[200];

    printf("Enter filename to append to: ");
    scanf("%s", filename);

    fp = fopen(filename, "a");   // open file in append mode
    if (fp == NULL) {
        printf("Error: Cannot open file!\n");
        return 1;
    }

    getchar();  // clear leftover newline from buffer

    printf("Enter text to append: ");
    fgets(text, sizeof(text), stdin);   // take full line input with spaces

    fprintf(fp, "%s", text);   // append text to file

    fclose(fp);

    printf("File updated successfully with appended text.\n");

    return 0;
}
