#include <stdio.h>

int main() {
    int n;
    printf("Enter size of array: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int visited[1000] = {0}; // marking array
    int repeated = -1;

    // ONE SINGLE LOOP
    for (int i = 0; i < n; i++) {
        if (visited[arr[i]] == 1) {
            repeated = arr[i];
            break;
        }
        visited[arr[i]] = 1;
    }

    printf("Repeated element: %d\n", repeated);

    return 0;
}
