#include <stdio.h>

int main(void) {
    int n;
    if (scanf("%d", &n) != 1 || n <= 0) return 0;
    int arr[n];

    for (int i = 0; i < n; ++i) {
        scanf("%d", &arr[i]);
    }

    for (int i = 0; i < n; ++i) {
        int pge = -1; // previous greater element
        // search to the left for nearest greater element
        for (int j = i - 1; j >= 0; --j) {
            if (arr[j] > arr[i]) { pge = arr[j]; break; }
        }

        if (i == n - 1)
            printf("%d", pge);
        else
            printf("%d, ", pge);
    }

    return 0;
}
