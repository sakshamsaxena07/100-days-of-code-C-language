//Write a program to implement a basic calculator using switch-case for +, -, *, /, %.

#include<stdio.h>
int main()
{
int num1,num2;
float result;
char ch;
printf("enter first number = ");
scanf("%d",&num1);
printf("enter second number = ");
scanf("%d",&num2);
printf("choose operator to perform operations = ");
scanf("%c",&ch);
result 0;
switch (ch)
{
case'+':
  result= num1 + num2;
  break;
case'-':
  result= num1 - num2;
  break;
case'*':
  result= num1 * num2;
  break;
case'/':
  result= num1 / num2;
  break;
default:
  printf("invalid operations\n);

}
printf("result: %d %c %d = %f\n" , num1,ch,num2,result);
return 0; 
}
