#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int nums[n], left[n], right[n], answer[n];

    // Read input
    for(int i = 0; i < n; i++)
        scanf("%d", &nums[i]);

    // Build prefix products (left)
    left[0] = 1;
    for(int i = 1; i < n; i++)
        left[i] = left[i-1] * nums[i-1];

    // Build suffix products (right)
    right[n-1] = 1;
    for(int i = n-2; i >= 0; i--)
        right[i] = right[i+1] * nums[i+1];

    // Multiply prefix and suffix products
    for(int i = 0; i < n; i++)
        answer[i] = left[i] * right[i];

    // Print result in required format
    printf("[");
    for(int i = 0; i < n; i++) {
        printf("%d", answer[i]);
        if(i != n-1) printf(",");
    }
    printf("]");

    return 0;
}
