#include <stdio.h>

int main() {
    char str[1000];
    int count = 0;

    // Read string (including spaces)
    fgets(str, sizeof(str), stdin);

    // Count characters until newline or null
    while(str[count] != '\0' && str[count] != '\n') {
        count++;
    }

    // Print result
    printf("%d", count);

    return 0;
}
