#include <stdio.h>

// Define enum with explicit starting value
enum Letters {
    A = 10,  // explicitly assigned 10
    B,       // auto-increments to 11
    C        // auto-increments to 12
};

int main() {
    printf("A=%d\n", A);
    printf("B=%d\n", B);
    printf("C=%d\n", C);

    return 0;
}
