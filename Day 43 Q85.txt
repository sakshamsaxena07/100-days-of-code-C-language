#include <stdio.h>

int main() {
    char str[1000];
    int len = 0;

    // Read string including spaces
    fgets(str, sizeof(str), stdin);

    // Find the length (without using strlen)
    while(str[len] != '\0' && str[len] != '\n') {
        len++;
    }

    // Print string in reverse
    for(int i = len - 1; i >= 0; i--) {
        printf("%c", str[i]);
    }

    return 0;
}
