#include <stdio.h>

int main() {
    char str[1000];
    int len = 0;

    // Read string
    fgets(str, sizeof(str), stdin);

    // Calculate length
    while(str[len] != '\0' && str[len] != '\n')
        len++;

    // Print all substrings
    for(int i = 0; i < len; i++) {
        for(int j = i; j < len; j++) {
            for(int k = i; k <= j; k++) {
                printf("%c", str[k]);
            }
            if(!(i == len-1 && j == len-1))
                printf(",");  // print comma except after last substring
        }
    }

    return 0;
}
