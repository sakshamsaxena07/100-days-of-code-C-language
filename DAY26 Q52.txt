/*
Write a program to print the following pattern:

*

*
*
*

*
*
*
*
*

*
*
*

*
*/

#include <stdio.h>

int main() {
    // Group 1: 1 star
    printf("*\n\n");

    // Group 2: 3 stars
    for(int i = 0; i < 3; i++) {
        printf("*\n");
    }
    printf("\n");

    // Group 3: 5 stars
    for(int i = 0; i < 5; i++) {
        printf("*\n");
    }
    printf("\n");

    // Group 4: 3 stars
    for(int i = 0; i < 3; i++) {
        printf("*\n");
    }
    printf("\n");

    // Group 5: 1 star
    printf("*\n");

    return 0;
}
