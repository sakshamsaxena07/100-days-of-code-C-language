#include <stdio.h>

int main() {
    int rows, cols;
    
    // Input the number of rows and columns
    scanf("%d %d", &rows, &cols);
    
    int matrix[rows][cols];  // declaring 2D array
    
    // Reading matrix elements
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
    
    // Printing matrix elements
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}
