#include <stdio.h>

int main() {
    int n, k;
    scanf("%d", &n);    // size of array
    
    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);   // input array
        
    scanf("%d", &k);    // window size
    
    // For each subarray of size k
    for(int i = 0; i <= n - k; i++) {
        int max = arr[i];   // assume first element is max
        
        for(int j = i + 1; j < i + k; j++) {
            if(arr[j] > max)
                max = arr[j];
        }
        
        printf("%d", max);
        if(i < n - k)   // print space except last value
            printf(" ");
    }
    
    return 0;
}
