#include <stdio.h>

int main() {
    char str[1000];
    int i = 0;

    // Read string including spaces
    fgets(str, sizeof(str), stdin);

    // Replace spaces with hyphens
    while(str[i] != '\0' && str[i] != '\n') {
        if(str[i] == ' ') {
            str[i] = '-';
        }
        i++;
    }

    // Print result
    printf("%s", str);

    return 0;
}
