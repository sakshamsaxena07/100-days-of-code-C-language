#include <stdio.h>

int main() {
    int n;
    scanf("%d %d", &n, &n);  // Matrix must be square

    int matrix[n][n];
    int distinct = 1; // assume true

    // Read matrix
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }

    // Check distinct diagonal elements
    for(int i = 0; i < n; i++) {
        for(int j = i + 1; j < n; j++) {
            if(matrix[i][i] == matrix[j][j]) {
                distinct = 0;
                break;
            }
        }
        if(!distinct) break;
    }

    // Print result
    if(distinct)
        printf("True");
    else
        printf("False");

    return 0;
}
