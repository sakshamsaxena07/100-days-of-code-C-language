#include <stdio.h>

int main() {
    int n, target;
    
    // Input array size
    printf("Enter number of elements in the array: ");
    scanf("%d", &n);
    
    int nums[n];
    
    // Input array elements
    printf("Enter the elements of the array:\n");
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }
    
    // Input target
    printf("Enter the target value: ");
    scanf("%d", &target);
    
    int found = 0; // Flag to check if solution exists
    
    // Check all pairs
    for(int i = 0; i < n-1; i++) {
        for(int j = i+1; j < n; j++) {
            if(nums[i] + nums[j] == target) {
                printf("%d %d\n", i, j);
                found = 1;
                break; // Exit inner loop after finding the pair
            }
        }
        if(found) break; // Exit outer loop if pair is found
    }
    
    if(!found) {
        printf("-1 -1\n");
    }
    
    return 0;
}
