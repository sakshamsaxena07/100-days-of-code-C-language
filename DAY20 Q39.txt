//Write a program to find the product of odd digits of a number.
#include <stdio.h>

int main() {
    int num, digit, product = 1;
    int hasOdd = 0;

    printf("Enter a number: ");
    scanf("%d", &num);

    // Process each digit
    while (num > 0) {
        digit = num % 10;

        if (digit % 2 != 0) {  // Check for odd digit
            product *= digit;
            hasOdd = 1; // Mark that at least one odd digit exists
        }

        num /= 10;
    }

    if (hasOdd)
        printf("Product of odd digits is %d\n", product);
    else
        printf("No odd digits found. Product is 0\n");

    return 0;
}
