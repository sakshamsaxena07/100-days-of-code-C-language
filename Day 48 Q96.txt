#include <stdio.h>

int main() {
    char str[1000];
    int i = 0, start = 0;

    // Read sentence
    fgets(str, sizeof(str), stdin);

    while (str[i] != '\0' && str[i] != '\n') {
        // If space found or end of string word-end reached
        if (str[i] == ' ') {
            // Reverse the word from 'start' to 'i-1'
            int left = start, right = i - 1;
            while (left < right) {
                char temp = str[left];
                str[left] = str[right];
                str[right] = temp;
                left++;
                right--;
            }
            start = i + 1;  // New word starts after space
        }
        i++;
    }

    // Reverse last word (since it won't end with space)
    int left = start, right = i - 1;
    while (left < right) {
        char temp = str[left];
        str[left] = str[right];
        str[right] = temp;
        left++;
        right--;
    }

    // Print result
    printf("%s", str);

    return 0;
}
