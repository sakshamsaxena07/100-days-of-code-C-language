#include <stdio.h>

int main() {
    char str[1000], ch;
    int count = 0, i = 0;

    // Read string
    fgets(str, sizeof(str), stdin);

    // Read character to search
    scanf("%c", &ch);

    // Count frequency
    while(str[i] != '\0' && str[i] != '\n') {
        if(str[i] == ch) {
            count++;
        }
        i++;
    }

    // Print result
    printf("%d", count);

    return 0;
}
