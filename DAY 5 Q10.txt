//Write a program to input time in seconds and convert it to hours:minutes:seconds format.

#include <stdio.h>

int main() 
{
    int sec, h, m, s;

    printf("Enter seconds:- ");
    scanf("%d", &sec);

    h = sec / 3600;
    sec = sec % 3600;
    m = sec / 60;
    s = sec % 60;

    printf("%d:%d:%d", h, m, s);

    return 0;
}
