#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    // Sum of 1 to n = n(n+1)/2
    int totalSum = n * (n + 1) / 2;

    // Try each possible pivot x
    int leftSum = 0;
    for(int x = 1; x <= n; x++) {
        leftSum += x;  // sum from 1 to x
        int rightSum = totalSum - (x - 1) * x / 2; // sum from x to n

        if(leftSum == rightSum) {
            printf("%d", x);
            return 0;
        }
    }

    printf("-1");  // No pivot found
    return 0;
}
