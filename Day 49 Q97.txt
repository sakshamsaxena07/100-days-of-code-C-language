#include <stdio.h>

int main() {
    char str[1000];
    int i = 0;

    // Read full name
    fgets(str, sizeof(str), stdin);

    // Print first initial (ignore spaces)
    while (str[i] == ' ') i++; // skip leading spaces
    if (str[i] != '\0' && str[i] != '\n')
        printf("%c.", str[i]);

    // Print initials after spaces
    while (str[i] != '\0' && str[i] != '\n') {
        if (str[i] == ' ' && str[i + 1] != ' ' && str[i + 1] != '\n') {
            printf("%c.", str[i + 1]);
        }
        i++;
    }

    return 0;
}
