#include <stdio.h>

// Define struct Student
struct Student {
    char name[50];
    int roll_no;
    int marks;
};

// Function to print student details
void printStudent(struct Student s) {
    printf("Name: %s | Roll: %d | Marks: %d\n", s.name, s.roll_no, s.marks);
}

int main() {
    struct Student s1;

    // Read student details
    printf("Enter Name: ");
    scanf("%s", s1.name);  // assuming no spaces

    printf("Enter Roll: ");
    scanf("%d", &s1.roll_no);

    printf("Enter Marks: ");
    scanf("%d", &s1.marks);

    // Call function to print details
    printStudent(s1);

    return 0;
}
