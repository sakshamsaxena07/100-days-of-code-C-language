#include <stdio.h>

int main() {
    char str[1000];
    int i = 0;

    // Read string (including spaces)
    fgets(str, sizeof(str), stdin);

    // Toggle case
    while(str[i] != '\0' && str[i] != '\n') {
        if(str[i] >= 'a' && str[i] <= 'z') {
            str[i] = str[i] - 32;  // lowercase to uppercase
        }
        else if(str[i] >= 'A' && str[i] <= 'Z') {
            str[i] = str[i] + 32;  // uppercase to lowercase
        }
        i++;
    }

    // Print toggled string
    printf("%s", str);

    return 0;
}
