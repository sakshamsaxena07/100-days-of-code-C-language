#include <stdio.h>

// Define enum
enum TrafficLight {
    RED,
    YELLOW,
    GREEN,
    TOTAL // helper to know number of enum items
};

int main() {
    // Array of names for printing
    const char *names[] = {"RED", "YELLOW", "GREEN"};

    // Loop through enum values
    for (int i = 0; i < TOTAL; i++) {
        printf("%s=%d\n", names[i], i);
    }

    return 0;
}
