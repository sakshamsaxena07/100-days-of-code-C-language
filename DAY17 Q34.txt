//Write a program to check if a number is prime.

#include<stdio.h>
int main()
{
 int i, count=0,n; 
 printf("enter a number");
 scanf("%d",&n);
 for (i=1; i<=n;i++)
 {
  if(n%i==0)
  count+=1;
 }
if(n==0||n==1)
  printf("%d is not prime",n);
else if(count>2)
  printf("%d is not prime",n);
else
  printf("%d is a prime number",n);
return 0;
}
