#include <stdio.h>

int main() {
    char str[1000];
    int freq[26] = {0};  // for lowercase a-z
    int i = 0;

    // Read string
    fgets(str, sizeof(str), stdin);

    // Traverse and find the first repeating lowercase letter
    while(str[i] != '\0' && str[i] != '\n') {
        char ch = str[i];

        if(ch >= 'a' && ch <= 'z') {
            if(freq[ch - 'a'] == 1) {  // already seen once
                printf("%c", ch);
                return 0;
            }
            freq[ch - 'a']++;
        }

        i++;
    }

    // If no repeating lowercase alphabet found
    printf("No repeating character");

    return 0;
}
