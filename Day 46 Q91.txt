#include <stdio.h>

int main() {
    char str[1000];
    int i = 0, j = 0;

    // Read string (including spaces)
    fgets(str, sizeof(str), stdin);

    // Process and remove vowels
    while(str[i] != '\0' && str[i] != '\n') {
        char ch = str[i];

        // If not a vowel, overwrite at same index
        if(!(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'||
             ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U')) {
            str[j++] = ch;
        }
        i++;
    }
    str[j] = '\0'; // terminate final string

    // Print result
    printf("%s", str);

    return 0;
}
