#include <stdio.h>

int main() {
    long long binary, temp;
    int digit;
    long long ones_complement = 0;
    long long place = 1;

    printf("Enter a binary number: ");
    scanf("%lld", &binary);

    temp = binary;

    while (temp > 0) {
        digit = temp % 10;

        if (digit == 0)
            digit = 1;
        else if (digit == 1)
            digit = 0;
        else {
            printf("Invalid binary number.\n");
            return 1; // Exit if digit not 0 or 1
        }

        ones_complement += digit * place;
        place *= 10;
        temp /= 10;
    }

    printf("1's complement: %lld\n", ones_complement);

    return 0;
}
