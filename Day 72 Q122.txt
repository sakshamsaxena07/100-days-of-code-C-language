#include <stdio.h>

int main() {
    FILE *fp;
    char line[200];

    fp = fopen("info.txt", "r");   // open file in read mode
    if (fp == NULL) {
        printf("Error: Cannot open file!\n");
        return 1;
    }

    // Read each line until EOF
    while (fgets(line, sizeof(line), fp) != NULL) {
        printf("%s", line);   // print each line
    }

    fclose(fp);   // close the file
    return 0;
}
