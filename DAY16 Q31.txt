//Write a program to take a number as input and print its equivalent binary representation.

#include <stdio.h>

int main() {
    int num, binary = 0, place = 1;

    printf("Enter a number: ");
    scanf("%d", &num);

    if (num == 0) {
        printf("Binary: 0\n");
        return 0;
    }

    while (num > 0) {
        int rem = num % 2;
        binary += rem * place;
        place *= 10;
        num /= 2;
    }

    printf("Binary: %d\n", binary);

    return 0;
}
