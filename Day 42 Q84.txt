#include <stdio.h>

int main() {
    char str[1000];
    int i = 0;

    // Read string with spaces
    fgets(str, sizeof(str), stdin);

    // Convert lowercase characters to uppercase
    while(str[i] != '\0' && str[i] != '\n') {
        if(str[i] >= 'a' && str[i] <= 'z') {
            str[i] = str[i] - 32;  // ASCII conversion
        }
        i++;
    }

    // Print the converted string
    printf("%s", str);

    return 0;
}
