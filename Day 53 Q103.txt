#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int nums[n];

    // Read array elements
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    int totalSum = 0, leftSum = 0;

    // Calculate total sum
    for(int i = 0; i < n; i++) {
        totalSum += nums[i];
    }

    // Find pivot index
    for(int i = 0; i < n; i++) {
        // Right sum = totalSum - leftSum - nums[i]
        if(leftSum == totalSum - leftSum - nums[i]) {
            printf("%d", i);
            return 0;
        }
        leftSum += nums[i];
    }

    // If no pivot found
    printf("-1");
    return 0;
}
