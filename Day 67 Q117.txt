#include <stdio.h>

int main() {
    int m, n;

    // Input sizes of the arrays
    printf("Enter size of first array: ");
    scanf("%d", &m);
    int nums1[m];
    printf("Enter elements of first sorted array:\n");
    for(int i = 0; i < m; i++) {
        scanf("%d", &nums1[i]);
    }

    printf("Enter size of second array: ");
    scanf("%d", &n);
    int nums2[n];
    printf("Enter elements of second sorted array:\n");
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums2[i]);
    }

    int merged[m + n]; // Array to store merged result
    int i = 0, j = 0, k = 0;

    // Merge arrays using two pointers
    while(i < m && j < n) {
        if(nums1[i] <= nums2[j]) {
            merged[k++] = nums1[i++];
        } else {
            merged[k++] = nums2[j++];
        }
    }

    // Copy remaining elements of nums1
    while(i < m) {
        merged[k++] = nums1[i++];
    }

    // Copy remaining elements of nums2
    while(j < n) {
        merged[k++] = nums2[j++];
    }

    // Print merged array
    printf("Merged sorted array:\n");
    for(int x = 0; x < m + n; x++) {
        printf("%d ", merged[x]);
    }
    printf("\n");

    return 0;
}
