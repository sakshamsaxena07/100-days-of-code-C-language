#include <stdio.h>

int main() {
    char str[1000];
    int i = 0;

    // Read string with spaces
    fgets(str, sizeof(str), stdin);

    // Print each character on a new line until newline or null
    while(str[i] != '\0' && str[i] != '\n') {
        printf("%c\n", str[i]);
        i++;
    }

    return 0;
}
