#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int nums[n];

    // Read array elements
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    // Boyerâ€“Moore Majority Vote Algorithm to find candidate
    int candidate = nums[0], count = 1;
    for(int i = 1; i < n; i++) {
        if(nums[i] == candidate)
            count++;
        else
            count--;

        if(count == 0) {
            candidate = nums[i];
            count = 1;
        }
    }

    // Check if candidate is actually majority
    count = 0;
    for(int i = 0; i < n; i++) {
        if(nums[i] == candidate)
            count++;
    }

    // Print result
    if(count > n/2)
        printf("%d", candidate);
    else
        printf("-1");

    return 0;
}
