#include <stdio.h>
#include <limits.h>

int main(void) {
    int n, k;
    if (scanf("%d", &n) != 1) return 0;      // read array size
    if (n <= 0) { printf("0"); return 0; }

    int arr[n];
    for (int i = 0; i < n; ++i) scanf("%d", &arr[i]);

    if (scanf("%d", &k) != 1) return 0;      // read k
    if (k <= 0 || k > n) { printf("0"); return 0; } // invalid k -> print 0

    long long windowSum = 0;
    for (int i = 0; i < k; ++i) windowSum += arr[i];

    long long maxSum = windowSum;

    for (int i = k; i < n; ++i) {
        windowSum += arr[i];
        windowSum -= arr[i - k];
        if (windowSum > maxSum) maxSum = windowSum;
    }

    printf("%lld", maxSum);
    return 0;
}
