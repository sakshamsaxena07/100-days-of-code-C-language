#include <stdio.h>

int main() {
    int n;
    scanf("%d %d", &n, &n);  // Since symmetric matrix must be square

    int matrix[n][n];
    int symmetric = 1; // assume true

    // Read matrix
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }

    // Check symmetry (A[i][j] == A[j][i])
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            if(matrix[i][j] != matrix[j][i]) {
                symmetric = 0;
                break;
            }
        }
        if(!symmetric) break;
    }

    // Output result
    if(symmetric)
        printf("True");
    else
        printf("False");

    return 0;
}
