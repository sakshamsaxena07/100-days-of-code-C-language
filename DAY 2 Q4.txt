//Write a program to calculate the area and circumference of a circle given its radius.

#include <stdio.h>
int main() 
{
    float r, area, circumference;
    printf("Enter radius:- ");
    scanf("%f", &r);

    area = 3.14 * r * r;
    circumference = 2 * 3.14 * r;

    printf("Area = %f\n", area);
    printf("Circumference = %f\n", circumference);

    return 0;
}
