#include <stdio.h>

int main() {
    int n, target;
    
    // Read array size
    scanf("%d", &n);
    int nums[n];
    
    // Read array elements
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }
    
    // Read target
    scanf("%d", &target);
    
    int first = -1, last = -1;
    
    // Find first occurrence
    for(int i = 0; i < n; i++) {
        if(nums[i] == target) {
            first = i;
            break;
        }
    }
    
    // Find last occurrence
    for(int i = n - 1; i >= 0; i--) {
        if(nums[i] == target) {
            last = i;
            break;
        }
    }
    
    printf("%d,%d", first, last);
    
    return 0;
}
